CREATE DATABASE IF NOT EXISTS ICT702_20031759_Individual_Assignment;
USE ICT702_20031759_Individual_Assignment;

START TRANSACTION;


CREATE TABLE MEMBER (
  MemberID INT AUTO_INCREMENT PRIMARY KEY,
  FirstName VARCHAR(50) NOT NULL,
  LastName VARCHAR(50) NOT NULL,
  Gender VARCHAR(10),
  DOB DATE,
  Phone VARCHAR(20),
  Email VARCHAR(100),
  Address VARCHAR(150)
);

CREATE TABLE MEMBERSHIPPLAN (
  PlanID INT AUTO_INCREMENT PRIMARY KEY,
  PlanName VARCHAR(50) NOT NULL,
  DurationMonths INT NOT NULL,
  Price DECIMAL(10,2) NOT NULL
);


CREATE TABLE MEMBERSHIP (
  MembershipID INT AUTO_INCREMENT PRIMARY KEY,
  MemberID INT NOT NULL,
  PlanID INT NOT NULL,
  StartDate DATE,
  EndDate DATE,
  Status VARCHAR(20),
  FOREIGN KEY (MemberID) REFERENCES MEMBER(MemberID),
  FOREIGN KEY (PlanID) REFERENCES MEMBERSHIPPLAN(PlanID)
);


CREATE TABLE STAFF (
  StaffID INT AUTO_INCREMENT PRIMARY KEY,
  Name VARCHAR(100) NOT NULL,
  HireDate DATE,
  Salary DECIMAL(10,2));


CREATE TABLE TRAINER (
  StaffID INT PRIMARY KEY,
  Speciality VARCHAR(100),
  Phone VARCHAR(20),
  FOREIGN KEY (StaffID) REFERENCES STAFF(StaffID)
);


CREATE TABLE RECEPTIONIST (
  StaffID INT PRIMARY KEY,
  Shift VARCHAR(50),
  FOREIGN KEY (StaffID) REFERENCES STAFF(StaffID));


CREATE TABLE CLASS (
  ClassID INT AUTO_INCREMENT PRIMARY KEY,
  ClassName VARCHAR(100) NOT NULL,
  DifficultyLevel VARCHAR(20),
  Capacity INT
);

-- CLASSSCHEDULE
CREATE TABLE CLASSSCHEDULE (
  ScheduleID INT AUTO_INCREMENT PRIMARY KEY,
  ClassID INT NOT NULL,
  TrainerID INT NOT NULL,
  ScheduledDate DATE,
  StartTime TIME,
  EndTime TIME,
  FOREIGN KEY (ClassID) REFERENCES CLASS(ClassID),
  FOREIGN KEY (TrainerID) REFERENCES TRAINER(StaffID)
);


CREATE TABLE BOOKING (
  BookingID INT AUTO_INCREMENT PRIMARY KEY,
  ScheduleID INT NOT NULL,
  MemberID INT NOT NULL,
  BookingDate DATE,
  Status VARCHAR(20),
  FOREIGN KEY (ScheduleID) REFERENCES CLASSSCHEDULE(ScheduleID),
  FOREIGN KEY (MemberID) REFERENCES MEMBER(MemberID)
);


CREATE TABLE EQUIPMENT (
  EquipmentID INT AUTO_INCREMENT PRIMARY KEY,
  EquipmentName VARCHAR(100) NOT NULL,
  PurchaseDate DATE,
  Condition VARCHAR(50)
);


CREATE TABLE MAINTENANCE (
  MaintenanceID INT AUTO_INCREMENT PRIMARY KEY,
  EquipmentID INT NOT NULL,
  StaffID INT NOT NULL,
  MaintenanceDate DATE,
  Description VARCHAR(200),
  FOREIGN KEY (EquipmentID) REFERENCES EQUIPMENT(EquipmentID),
  FOREIGN KEY (StaffID) REFERENCES STAFF(StaffID)  );


CREATE TABLE PAYMENT (
  PaymentID INT AUTO_INCREMENT PRIMARY KEY,
  MembershipID INT NOT NULL,
  PaymentDate DATE,
  Amount DECIMAL(10,2),
  Method VARCHAR(50),
  FOREIGN KEY (MembershipID) REFERENCES MEMBERSHIP(MembershipID)
);
